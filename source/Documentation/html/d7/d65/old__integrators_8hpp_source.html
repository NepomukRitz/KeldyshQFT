30template <typename Q, typename Integrand> auto integrator_riemann(const Integrand& integrand, int N) -> Q {
31 rvec spacings(N);
32 vec<Q> integrand_values(N);
33 double w0, w1, w2;
34
35 spacings[0] = bfreqs[1] - bfreqs[0];
36 integrand_values[0] = integrand(bfreqs[0]);
37
38 for (int i=1; i<N-1; ++i) {
39 w0 = bfreqs[i-1]; // now specifically relies on bfreqs, only works if bfreqs = ffreqs...
40 w1 = bfreqs[i];
41 w2 = bfreqs[i+1];
42 spacings[i] = w2 - w0;
43 integrand_values[i] = integrand(w1);
44 }
45
46 spacings[N-1] = bfreqs[N-1] - bfreqs[N-2];
47 integrand_values[N-1] = integrand(bfreqs[N-1]);
48
49 return 1/2.*dotproduct<Q>(integrand_values, spacings);
50}

61template <typename Q, typename Integrand> auto integrator_simpson(const Integrand& integrand, double a, double b, int N) -> Q {
62 if (N < 3) N = 3;
63 double dx = (b-a)/((double)(N-1));
64 rvec simpson(N);
65 vec<Q> integrand_values(N);
66
67//#pragma acc parallel loop private (i)
68 for (int i=0; i<N; ++i)
69 {
70 integrand_values[i] = integrand(a+i*dx);
71 simpson[i] = 2. +2*(i%2);
72 }
73 simpson[0] = 1.;
74 simpson[N-1]=1.;
75

86template <typename Q, typename Integrand> auto integrator_simpson(const Integrand& integrand, double a, double b, double w, int N) -> Q {
87 Q result = 0.; // initialize result
88
89 const double glb_Gamma = 1.;
90 double Delta = 2*glb_Gamma; // half-width of the region around the sharp feature which should get better resolution
91 int Nw = N/5; // number of additional points around the sharp feature
92 if (!(Nw % 2)) Nw += 1; // number of points needs to be odd
93
94
95 int Na = (int)(N * (w - a) / (b - a)); // number of points in the first interval (fraction of N)
96 if (!(Na % 2)) Na += 1; // number of points needs to be odd
97 int Nb = (int)(N * (b - w) / (b - a)); // number of points in the first interval (fraction of N)
98 if (!(Nb % 2)) Nb += 1; // number of points needs to be odd
99
100 // Compute different contributions
101 result += integrator_simpson<Q>(integrand, a, w-Delta, Na); // interval of frequencies < w
102 result += integrator_simpson<Q>(integrand, w-Delta, w+Delta, Nw); // interval around w
103 result += integrator_simpson<Q>(integrand, w+Delta, b, Nb); // interval of frequencies > w
104 return result;
105}

114template <typename Q, typename Integrand> auto integrator_simpson(const Integrand& integrand, double a, double b, double w1_in, double w2_in, int N) -> Q {
115
116 Q result = 0.; // initialize result
117 int Na, Nb, Nc; // number of points in different intervals
118 const double glb_T = 1.;
119
120 double Delta = 5*glb_T; // half-width of the region around the sharp features which should get better resolution
121 int Nw = N/10; // number of additional points around each of the sharp features
122 if (!(Nw % 2)) Nw += 1; // number of points needs to be odd
123
124 // Order the frequencies w1/w2 such that always w1 < w2.
125 double w1, w2;
126 if (w1_in < w2_in) {
127 w1 = w1_in; w2 = w2_in;
128 }
129 else {
130 w1 = w2_in; w2 = w1_in;
131 }
132
133 // First case: w1/w2 are away from boundary (assume that b-w2 == w1-a)
134 if (b-w2 > Delta) {
135 // First compute contributions of intervals [a,w1-Delta] and [w2+Delta,b]
136 Na = (int)(N * (w1 - a) / (b - a)); // number of points in the first interval (fraction of N)
137 if (!(Na % 2)) Na += 1; // number of points needs to be odd
138 Nb = (int)(N * (b - w2) / (b - a)); // number of points in the last interval
139 if (!(Nb % 2)) Nb += 1; // number of points needs to be odd
140
141 result += integrator_simpson<Q>(integrand, a, w1-Delta, Na);
142 result += integrator_simpson<Q>(integrand, w2+Delta, b, Nb);
143
144 // Check if w1 and w2 are too close to consider them separately
145 if (w2-w1 > 10*Delta) { // w1 and w2 are far enough away from each other to consider separate regions around them
146 Nc = (int)(N * (w2 - w1)/(b - a)); // number of points in the central interval between w1/w2
147 if (!(Nc % 2)) Nc += 1; // number of points needs to be odd
148
149 // Compute contributions of the intervals around w1/w2 and between them
150 result += integrator_simpson<Q>(integrand, w1-Delta, w1+Delta, Nw);
151 result += integrator_simpson<Q>(integrand, w1+Delta, w2-Delta, Nc);
152 result += integrator_simpson<Q>(integrand, w2-Delta, w2+Delta, Nw);
153 }
154 else { // w1/w2 are close to each other only consider single contribution
155 Nc = (int)(Nw * (w2-w1+2*Delta)/(2*Delta)); // number of points in the interval enclosing w1/w2
156 // (depends on the distance between w1/w2)
157 if (!(Nc % 2)) Nc += 1; // number of points needs to be odd
158
159 result += integrator_simpson<Q>(integrand, w1-Delta, w2+Delta, Nc);
160 }
161 }
162
163 // Second case: w1/w2 are close to boundaries a/b // currently not necessary since w1/2 = +-w/2
164 else {
165 result += integrator_simpson<Q>(integrand, a, w1+Delta, Nw); // interval around w1/a
166 result += integrator_simpson<Q>(integrand, w1+Delta, w2-Delta, N); // interval between w1/w2
167 result += integrator_simpson<Q>(integrand, w2-Delta, b, Nw); // interval around w2/b
168 }
169
170 return result;
171}

174// compute Simpson rule for 3 given values and given step size dx
175template <typename Q>
176auto simpson_rule_3(Q& val0, Q& val1, Q& val2, double& dx) -> Q {
177 return dx / 3. * (val0 + 4.*val1 + val2);
178}

180// compute Simpson rule for vector of 3 given values and given step size dx
181template <typename Q>
182auto simpson_rule_3(vec<Q>& values, double& dx) -> Q {
183 return dx / 3. * (values[0] + 4.*values[1] + values[2]);
