202template <typename Q>
203auto Propagator<Q>::GK(const double v, const int i_in) const -> Q
204{
205 if constexpr (EQUILIBRIUM and not (REG==5)) {
206 // FDT in equilibrium: (1-2*Eff_distr)*(GR-GA)
207 //return (1.-2.*Eff_distr(v))*(GR(v, i_in) - GA(v, i_in));
208 return glb_i * (Eff_fac(v,T) * 2. * myimag(GR(v, i_in))); // more efficient: only one interpolation instead of two
209 }
210 else if constexpr (EQUILIBRIUM and (REG==5)){
211 // evaluate FDT with T=Lambda
212 //return glb_i * (Eff_fac(v,Lambda) * 2. * myimag(GR(v, i_in))); // explicitly uses FDT
213 return (selfenergy.valsmooth(1, v, i_in) - glb_i * Gamma * Eff_fac(v, Lambda)) * std::norm(GR(v, i_in));
214 }
215 else {
216 // General form (Dyson equation): GR*(SigmaK+SigmaK_res)*GA
217 // Derivation of equilibrium form:
218 // \Sigma^K = (1-2n_F)(\Sigma^R-\Sigma^A), accordingly for \Sigma^K_res
219 // \Rightarrow G^K = (1-2n_F) G^R G^A [ (\Sigma+\Sigma_res)^R - (\Sigma+\Sigma_res)^A ]
220 // = (1-2n_F) G^R G^A [ (G^A)^{-1} - (G^R)^{-1} ] = (1-2n_F) (G^R-G^A)
221 // note that \Sigma_res^R = - i (Gamma+Lambda) / 2.
222 // return GR(v, i_in) * (selfenergy.valsmooth(1, v, i_in) - glb_i*(Gamma+Lambda)*(1.-2.*Eff_distr(v))) * GA(v, i_in);
223 // more efficient: only one interpolation instead of two; std::norm(c)=std::std::abs(c)^2
224 return std::norm(GR(v, i_in)) * (selfenergy.valsmooth(1, v, i_in) - glb_i * ((Gamma + Lambda) * Eff_fac(v,T)));
225 }
226}

242template <typename Q>
243auto Propagator<Q>::SK(const double v, const int i_in) const -> Q
244{
245 if constexpr (EQUILIBRIUM and not (REG==5)) {
246 // FDT in equilibrium: (1-2*Eff_distr)*(SR-SA)
247 //return (1.-2.*Eff_distr(v))*(SR(v, i_in) - myconj(SR(v, i_in)));
248 return glb_i * (Eff_fac(v,T) * 2. * myimag(SR(v, i_in)));
249 }
250 else if constexpr (EQUILIBRIUM and (REG==5)){
251 // special form of SK in the temperature flow
252 const double root_denominator = Lambda * cosh(v/(2.0*Lambda));
253 // return -glb_i * v * myimag(GR_REG5(v, i_in)) / (root_denominator*root_denominator); //explicitly uses FDT
254 return glb_i * 0.5 * Gamma * v * std::norm(GR(v, i_in)) / (root_denominator*root_denominator) ;
255 }
256 else {
257 // Derivation of general matrix form:
258 // S = - G * ( \partial_\Lambda G_0^{-1} ) * G
259 // where G = (0, G^A; G^R, G^K), G_0^{-1} = (Ginv0K, G_0^{A,-1}; G_0^{R,-1}, 0), Ginv0K = - \Sigma^K_res, \dot{Ginv0K} = i (1-2n_F)
260 // Thus, S^K = - G^R \dot{G_0^{R,-1}} G^K - G^K \dot{G_0^{A,-1}} G^A - G^R Ginv0K G^A
261 // Upon inserting G^K = (1-2n_F) (G^R-G^A), one recovers the equilibrium formula
262 //Q retarded = -0.5*glb_i*GR(v, i_in)*GK(v, i_in);
263 //Q advanced = +0.5*glb_i*GK(v, i_in)*GA(v, i_in);
264 //Q extra = -glb_i*(1.-2.*Eff_distr(v))*GR(v, i_in)*GA(v, i_in);
265 //return retarded + advanced + extra;
266 //return GK(v, i_in)*imag(GR(v, i_in)) - glb_i*(1.-2.*Eff_distr(v))*GR(v, i_in)*GA(v, i_in); // more efficient
267 //return GK(v, i_in)*imag(GR(v, i_in)) - glb_i*(Eff_fac(v)) * std::norm( GR(v, i_in) ); // more efficient
268 // most efficient: insert GK, factor out, combine real factors
269 Q gr = GR(v, i_in);
270 double gri = myimag(gr);
271 double grn = std::norm(gr);
272 return selfenergy.valsmooth(1, v, i_in) * (grn * gri) -
273 glb_i * (grn * Eff_fac(v,T) * (1. + (Gamma + Lambda) * gri));
274 }

315template <typename Q>
316auto Propagator<Q>::valsmooth(const int iK, const freqType v, const int i_in) const -> Q {
317 switch (type){
318 case 'g' : //Good ol' regular propagator
319 if constexpr (KELDYSH){
320 if constexpr(CONTOUR_BASIS != 1) {
321 switch (iK) {
322 case 0:
323 return GR(v, i_in);
324 case 1:
325 return GK(v, i_in);
326 default:
327 utils::print("ERROR! Invalid Keldysh index. Abort.");
328 assert(false);
329 }
330 }
331 else {
332 const Q GR_ = GR(v, i_in);
333 const Q GA_ = conj(GR_);
334 const Q GK_ = GK(v, i_in);
335 switch (iK){
336 case 0:
337 return 0.5*(GA_ + GR_ + GK_);
338 case 1:
339 return 0.5*(GA_ - GR_ + GK_);
340 case 2:
341 return 0.5*(-GA_ + GR_ + GK_);
342 case 3:
343 return 0.5*(-GA_ - GR_ + GK_);
344 default:
345 utils::print("ERROR! Invalid Keldysh index. Abort.");
346 assert(false);
347 }
348 }
349 }
350 else{
351 return GM(v, i_in);
352 }
353 break;
354 case 's':
355 if constexpr (KELDYSH){
356 if constexpr(CONTOUR_BASIS != 1) {
357 // Keldysh basis:
358 switch (iK){
359 case 0:
360 return SR(v, i_in);
361 case 1:
362 return SK(v, i_in);
363 default:
364 utils::print("ERROR! Invalid Keldysh index. Abort.");
365 assert(false);
366 }
367 }
368 else {
369 // Contour basis:
370 const Q SR_ = SR(v, i_in);
371 const Q SA_ = conj(SR_);
372 const Q SK_ = SK(v, i_in);
373 switch (iK){
374 case 0:
375 return 0.5*(SA_ + SR_ + SK_);
376 case 1:
377 return 0.5*(SA_ - SR_ + SK_);
378 case 2:
379 return 0.5*(-SA_ + SR_ + SK_);
380 case 3:
381 return 0.5*(-SA_ - SR_ + SK_);
382 default:
383 utils::print("ERROR! Invalid Keldysh index. Abort.");
384 assert(false);
385 }
386 }
387 }
388 else{
389 return SM(v, i_in);
390 }
391 break;
392 case 'k': // including the Katanin extension
393 if constexpr (KELDYSH){
394 if constexpr(CONTOUR_BASIS != 1) {
395 switch (iK) {
396 case 0:
397 return SR(v, i_in) + Katanin_R(v, i_in);
398 case 1:
399 return SK(v, i_in) + Katanin_K(v, i_in);
400 default:
401 utils::print("ERROR! Invalid Keldysh index. Abort.");
402 assert(false);
403 }
404 }
405 else {
406 const Q SR_ = SR(v, i_in) + Katanin_R(v, i_in);
407 const Q SA_ = conj(SR_);
408 const Q SK_ = SK(v, i_in) + Katanin_K(v, i_in);
409 switch (iK){
410 case 0: return 0.5*(SA_ + SR_ + SK_);
411 case 1: return 0.5*(SA_ - SR_ + SK_);
412 case 2: return 0.5*(-SA_ + SR_ + SK_);
413 case 3: return 0.5*(-SA_ - SR_ + SK_);
414 default:
415 utils::print("ERROR! Invalid Keldysh index. Abort.");
416 assert(false);
417 }
418 }
419 }
420 else{
421 return SM(v, i_in)
422 + GM(v, i_in) * diff_selfenergy.valsmooth(0, v, i_in) * GM(v, i_in);
423 }
424 break;
425 case 'e': // purely the Katanin extension
426 if constexpr (KELDYSH){
427 if constexpr(CONTOUR_BASIS != 1) {
428 switch (iK) {
429 case 0: return Katanin_R(v, i_in);
430 case 1: return Katanin_K(v, i_in);
431 default:
432 utils::print("ERROR! Invalid Keldysh index. Abort.");
433 assert(false);
434 }
435 }
436 else {
437 const Q SR_ = Katanin_R(v, i_in);
438 const Q SA_ = conj(SR_);
439 const Q SK_ = Katanin_K(v, i_in);
440 switch (iK){
441 case 0: return 0.5*(SA_ + SR_ + SK_);
442 case 1: return 0.5*(SA_ - SR_ + SK_);
443 case 2: return 0.5*(-SA_ + SR_ + SK_);
444 case 3: return 0.5*(-SA_ - SR_ + SK_);
445 default:
446 utils::print("ERROR! Invalid Keldysh index. Abort.");
447 assert(false);
448 }
449 }
450 }
451 else{
452 return GM(v, i_in) * diff_selfenergy.valsmooth(0, v, i_in) * GM(v, i_in);
453 }
454 break;
455 default:
456 utils::print("ERROR! Invalid Keldysh index. Abort.");
457 assert(false);
458 exit(1); // Failure
459 }
460 assert(false);
461 return 0.;
462}

597template <typename Q>
598auto Propagator<Q>::norm() const -> double {
599 double out = 0.;
600 for (int i = 0; i < nPROP; i++) {
601 if (KELDYSH) out += pow(std::abs(GR(selfenergy.Sigma.frequencies.primary_grid.get_frequency(i), 0)), 2.);
602 else out += pow(std::abs(GM(selfenergy.Sigma.frequencies.primary_grid.get_frequency(i), 0)), 2.);
603 }
604 return sqrt(out);
605}
